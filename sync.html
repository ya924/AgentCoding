步驟 2：於 Supabase SQL Editor 執行建表語法

在開始同步前，請前往您的 Supabase 後台 SQL Editor 貼上並執行以下語法建立資料表： 

-- 建立 github_repos 資料表
create table if not exists public.github_repos (
    id bigint primary key,
    name text not null,
    full_name text not null,
    html_url text not null,
    description text,
    language text,
    stargazers_count integer default 0,
    forks_count integer default 0,
    pushed_at timestamptz,
    custom_class text,
    created_at timestamptz default now()
);

-- 啟用 Row Level Security (RLS)
alter table public.github_repos enable row level security;

-- 允許所有人查詢 (Select)
create policy "Allow public read access"
on public.github_repos for select
using (true);

-- 允許所有人插入與更新 (Upsert)
create policy "Allow public upsert access"
on public.github_repos for insert
with check (true);

create policy "Allow public update access"
on public.github_repos for update
using (true);
